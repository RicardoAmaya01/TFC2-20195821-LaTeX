// Estructura simplificada del detector de latidos
bool update(long ir) {
  // Suavizado de señal-Promedio movil
  buf[idx] = ir;
  idx = (idx + 1) % N;
  float smooth = 0;
  for (int i = 0; i < N; i++) smooth += buf[i];
  smooth /= N;



  // Deteccion dinamica de umbral (
  maxIR = max(maxIR, smooth);
  minIR = min(minIR, smooth);
  float thr = minIR + (maxIR - minIR) * 0.45f; 

  // Deteccion de flanco de subida
  if (smooth > thr && smooth > last) rising = true;

  //  Confirmacion de latido y calculo de BPM
  if (rising && smooth < last) { // Cruce por cero (pico encontrado)
     unsigned long now = millis();
     unsigned long interval = now - lastBeat;
     
     // Filtrado de falsos positivos (Rango 40-140 BPM)
     float bpmCalc = 60000.0f / interval;
     if (bpmCalc >= 40 && bpmCalc <= 140) {
        addRR(interval); // Guardar intervalo R-R para FFT
        return true;     // Latido valido
     }
     lastBeat = now;
     rising = false;
  }
  last = smooth;
  return false;
}

bool computeLFHF(float &LF, float &HF, float &ratio) {
  //Interpolacion de intervalos R-R 
  int N = interpolateRR(); 
  if (N < 32) return false; 

  for (int i = 0; i < FFT_SIZE; i++) {
    vReal[i] = (i < N) ? rrInterp[i] : 0;
    vImag[i] = 0;
  }

  // Transformada de Fourier
  FFT.dcRemoval(); 
  FFT.windowing(FFTWindow::Hamming, FFTDirection::Forward);
  FFT.compute(FFTDirection::Forward);
  FFT.complexToMagnitude();

  // Integracion de bandas de frecuencia (LF y HF)
  float df = RESAMPLE_HZ / FFT_SIZE; // Resolucion de frecuencia
  LF = HF = 0;

  for (int i = 1; i < FFT_SIZE / 2; i++) {
    float f = i * df;
    // Banda Baja Frecuencia : 0.04 - 0.15 Hz
    if (f >= 0.04 && f <= 0.15) LF += vReal[i];
    // Banda Alta Frecuencia: 0.15 - 0.40 Hz
    else if (f >= 0.15 && f <= 0.40) HF += vReal[i];
  }

 
  if (HF < 1e-4) HF = 1e-4; 
  ratio = LF / HF;
  return true;
}
